36[[nodiscard]] std::vector<std::byte> compressData(std::span<const std::byte> data, int16_t level, CompressionMethod method) {
37 switch (method) {
38 using enum CompressionMethod;
39 case DEFLATE: {
40 mz_ulong compressedSize = mz_compressBound(data.size());
41 std::vector<std::byte> out(compressedSize);
42
43 int status = MZ_OK;
44 while ((status = mz_compress2(reinterpret_cast<unsigned char*>(out.data()), &compressedSize, reinterpret_cast<const unsigned char*>(data.data()), data.size(), level)) == MZ_BUF_ERROR) {
45 compressedSize *= 2;
46 out.resize(compressedSize);
47 }
48
49 if (status != MZ_OK) {
50 return {};
51 }
52 out.resize(compressedSize);
53 return out;
54 }
55 case ZSTD: {
56 if (level < 0) {
57 level = 6;
58 }
59
60 const auto expectedSize = ZSTD_compressBound(data.size());
61 std::vector<std::byte> out(expectedSize);
62
63 const auto compressedSize = ZSTD_compress(out.data(), expectedSize, data.data(), data.size(), level);
64 if (ZSTD_isError(compressedSize)) {
65 return {};
66 }
67
68 out.resize(compressedSize);
69 return out;
70 }
71 case CONSOLE_LZMA: {
72 if (const auto out = compression::compressValveLZMA(data, level)) {
73 return *out;
74 }
75 return {};
76 }
77 }
78 return {};
79}

81template<std::unsigned_integral T, bool ExistingDataIsSwizzled>
82constexpr void swizzleUncompressedImageData(std::span<std::byte> inputData, std::span<std::byte> outputData, ImageFormat format, uint16_t width, uint16_t height, uint16_t depth) {
83 width *= ImageFormatDetails::bpp(format) / (sizeof(T) * 8);
84 const auto zIndex = [
85 widthL2 = static_cast<int>(math::log2ceil(width)),
86 heightL2 = static_cast<int>(math::log2ceil(height)),
87 depthL2 = static_cast<int>(math::log2ceil(depth))
88 ](uint32_t x, uint32_t y, uint32_t z) {
89 auto widthL2m = widthL2;
90 auto heightL2m = heightL2;
91 auto depthL2m = depthL2;
92 uint32_t offset = 0;
93 uint32_t shiftCount = 0;
94 do {
95 if (depthL2m --> 0) {
96 offset |= (z & 1) << shiftCount++;
97 z >>= 1;
98 }
99 if (heightL2m --> 0) {
100 offset |= (y & 1) << shiftCount++;
101 y >>= 1;
102 }
103 if (widthL2m --> 0) {
104 offset |= (x & 1) << shiftCount++;
105 x >>= 1;
106 }
107 } while (x || y || z);
108 return offset;
109 };
110
111 const auto* inputPtr = reinterpret_cast<const T*>(inputData.data());
112 auto* outputPtr = reinterpret_cast<T*>(outputData.data());
113 for (uint16_t x = 0; x < width; x++) {
114 for (uint16_t y = 0; y < height; y++) {
115 for (uint16_t z = 0; z < depth; z++) {
116 if constexpr (ExistingDataIsSwizzled) {
117 *outputPtr++ = reinterpret_cast<const T*>(inputData.data())[zIndex(x, y, z)];
118 } else {
119 reinterpret_cast<T*>(outputData.data())[zIndex(x, y, z)] = *inputPtr++;
120 }
121 }
122 }
123 }

213 for (int k = 0; k < faceCount; k++) {
214 for (int i = 0; i < mipCount; i++) {

238Resource::ConvertedData Resource::convertData() const {
239 switch (this->type) {
240 case TYPE_PARTICLE_SHEET_DATA:
241 if (this->data.size() <= sizeof(uint32_t)) {
242 return {};
243 }
244 return SHT{{reinterpret_cast<const std::byte*>(this->data.data()) + sizeof(uint32_t), *reinterpret_cast<const uint32_t*>(this->data.data())}};
245 case TYPE_CRC:
246 case TYPE_EXTENDED_FLAGS:
247 if (this->data.size() != sizeof(uint32_t)) {
248 return {};
249 }
250 return *reinterpret_cast<const uint32_t*>(this->data.data());
251 case TYPE_LOD_CONTROL_INFO:
252 if (this->data.size() != sizeof(uint32_t)) {
253 return {};
254 }
255 return std::make_tuple(
256 *(reinterpret_cast<const uint8_t*>(this->data.data()) + 0),
257 *(reinterpret_cast<const uint8_t*>(this->data.data()) + 1),
258 *(reinterpret_cast<const uint8_t*>(this->data.data()) + 2),
259 *(reinterpret_cast<const uint8_t*>(this->data.data()) + 3));
260 case TYPE_KEYVALUES_DATA:
261 if (this->data.size() <= sizeof(uint32_t)) {
262 return "";
263 }
264 return std::string(reinterpret_cast<const char*>(this->data.data()) + sizeof(uint32_t), *reinterpret_cast<const uint32_t*>(this->data.data()));
265 case TYPE_HOTSPOT_DATA:
266 if (this->data.size() <= sizeof(uint32_t)) {
267 return {};
268 }
269 return HOT{{reinterpret_cast<const std::byte*>(this->data.data()) + sizeof(uint32_t), *reinterpret_cast<const uint32_t*>(this->data.data())}};
270 default:
271 break;
272 }
273 return {};
274}

595 return;
596 }
597 ::swapImageDataEndianForConsole<true>(reorderedFallbackData, this->format, this->fallbackMipCount, this->frameCount, faceCount, this->fallbackWidth, this->fallbackHeight, 1, this->platform);
598 }
599
600 this->opened = headerSizeIsAccurate;
601 if (parseHeaderOnly) {
602 return;
603 }
604
605 const auto imageSize = ImageFormatDetails::getDataLengthXBOX(true, this->format, this->mipCount, this->frameCount, faceCount, this->width, this->height, this->depth);
606 std::vector<std::byte> reorderedImageData;
607 if (this->hasImageData()) {
608 reorderedImageData = ::convertBetweenDDSAndVTFMipOrderForXBOX<true>(true, stream.seek(imageOffset).read_span<std::byte>(imageSize), this->format, this->mipCount, this->frameCount, faceCount, this->width, this->height, this->depth, ok);
609 if (!ok) {
610 this->opened = false;
611 return;
612 }
613 ::swapImageDataEndianForConsole<true>(reorderedImageData, this->format, this->mipCount, this->frameCount, faceCount, this->width, this->height, this->depth, this->platform);
614 }
615
616 // By this point we cannot use spans over data, it will change here
617 if (this->hasFallbackData()) {
618 this->setResourceInternal(Resource::TYPE_FALLBACK_DATA, reorderedFallbackData);
619 }
620 if (this->hasImageData()) {
621 this->setResourceInternal(Resource::TYPE_IMAGE_DATA, reorderedImageData);
622 }
623 return;
624 }
625 }
626 case PLATFORM_X360:
627 case PLATFORM_PS3_ORANGEBOX:
628 case PLATFORM_PS3_PORTAL2: {
629 uint8_t resourceCount;
630 stream
631 .read(this->flags)
632 .read(this->width)
633 .read(this->height)
634 .read(this->depth)
635 .read(this->frameCount)
636 .skip<uint16_t>() // preload
637 .read(this->consoleMipScale)
638 .read(resourceCount)
639 .read(this->reflectivity[0])
640 .read(this->reflectivity[1])
641 .read(this->reflectivity[2])
642 .read(this->bumpMapScale)
643 .read(this->format)
644 .skip<math::Vec4ui8>() // lowResImageSample (replacement for thumbnail resource, linear color pixel)
645 .skip<uint32_t>(); // compressedLength
646
647 postReadTransform();
648
649 // Align to 16 bytes
650 if (this->platform == PLATFORM_PS3_PORTAL2) {
651 stream.skip<uint32_t>();
652 }
653
654 this->mipCount = (this->flags & FLAG_V0_NO_MIP) ? 1 : ImageDimensions::getMaximumMipCount(this->width, this->height, this->depth);
655
656 if (parseHeaderOnly) {
657 this->opened = true;
658 return;
659 }
660
661 this->resources.reserve(resourceCount);
662 readResources(resourceCount);
663
664 this->opened = stream.tell() == headerSize;
665
666 // The resources vector isn't modified by setResourceInternal when we're not adding a new one, so this is fine
667 for (const auto& resource : this->resources) {
668 // Decompress LZMA resources
669 if (BufferStreamReadOnly rsrcStream{resource.data.data(), resource.data.size()}; rsrcStream.read<uint32_t>() == compression::VALVE_LZMA_SIGNATURE) {
670 if (auto decompressedData = compression::decompressValveLZMA(resource.data)) {
671 this->setResourceInternal(resource.type, *decompressedData);
672
673 if (resource.type == Resource::TYPE_IMAGE_DATA) {
674 // Do this here because compressionLength in header can be garbage on PS3 orange box
675 this->compressionMethod = CompressionMethod::CONSOLE_LZMA;
676 }
677 }
678 }
679
680 if (resource.type == Resource::TYPE_THUMBNAIL_DATA) {
681 ::swapImageDataEndianForConsole<true>(resource.data, this->thumbnailFormat, 1, 1, 1, this->thumbnailWidth, this->thumbnailHeight, 1, this->platform);
682 } else if (resource.type == Resource::TYPE_IMAGE_DATA) {
683 ::swapImageDataEndianForConsole<true>(resource.data, this->format, this->mipCount, this->frameCount, this->getFaceCount(), this->width, this->height, this->depth, this->platform);
684 } else if (!(resource.flags & Resource::FLAG_LOCAL_DATA) && resource.data.size() >= sizeof(uint32_t)) {
685 BufferStream::swap_endian(reinterpret_cast<uint32_t*>(resource.data.data()));
686 }
687 }
688 return;
689 }
690 }
691}

746bool VTF::createInternal(VTF& writer, CreationOptions options) {
747 bool out = true;
748 if (writer.hasImageData() && (options.invertGreenChannel || options.gammaCorrection != 1.f)) {
749 for (int i = 0; i < writer.mipCount; i++) {
750 for (int j = 0; j < writer.frameCount; j++) {
751 for (int k = 0; k < writer.getFaceCount(); k++) {
752 for (int l = 0; l < writer.depth; l++) {
753 if (options.invertGreenChannel && !writer.setImage(ImageConversion::invertGreenChannelForImageData(writer.getImageDataRaw(i, j, k, l), writer.getFormat(), writer.getWidth(i), writer.getHeight(i)), writer.getFormat(), writer.getWidth(i), writer.getHeight(i), ImageConversion::ResizeFilter::DEFAULT, i, j, k, l)) {
754 out = false;
755 }
756 if (options.gammaCorrection != 1.f && !writer.setImage(ImageConversion::gammaCorrectImageData(writer.getImageDataRaw(i, j, k, l), writer.getFormat(), writer.getWidth(i), writer.getHeight(i), options.gammaCorrection), writer.getFormat(), writer.getWidth(i), writer.getHeight(i), ImageConversion::ResizeFilter::DEFAULT, i, j, k, l)) {
757 out = false;
758 }
759 }
760 }
761 }
762 }
763 }
764 writer.setPlatform(options.platform);
765 if (options.computeReflectivity) {
766 writer.computeReflectivity();
767 }
768 if (options.initialFrameCount > 1 || options.isCubeMap || options.initialDepth > 1) {
769 if (!writer.setFrameFaceAndDepth(options.initialFrameCount, options.isCubeMap, options.initialDepth)) {
770 out = false;
771 }
772 }
773 writer.setStartFrame(options.startFrame);
774 writer.setBumpMapScale(options.bumpMapScale);
775 if (options.computeThumbnail) {
776 writer.computeThumbnail();
777 }
778 if (options.outputFormat == FORMAT_UNCHANGED) {
779 options.outputFormat = writer.getFormat();
780 } else if (options.outputFormat == FORMAT_DEFAULT) {
781 options.outputFormat = VTF::getDefaultCompressedFormat(writer.getFormat(), writer.getVersion(), options.isCubeMap);
782 }
783 if (options.computeMips) {
784 if (const auto maxMipCount = ImageDimensions::getMaximumMipCount(writer.getWidth(), writer.getHeight(), writer.getDepth()); maxMipCount > 1) {
785 if (!writer.setMipCount(maxMipCount)) {
786 out = false;
787 }
788 writer.computeMips(options.filter);
789 }
790 }
791 writer.setFormat(options.outputFormat, ImageConversion::ResizeFilter::DEFAULT, options.compressedFormatQuality);
792 if (options.computeTransparencyFlags) {
793 writer.computeTransparencyFlags();
794 }
795 writer.setCompressionLevel(options.compressionLevel);
796 writer.setCompressionMethod(options.compressionMethod);
797 writer.setConsoleMipScale(options.consoleMipScale);
798 return out;
799}

866 if (this->platform == newPlatform) {
867 return;
868 }
869
870 // hack to allow VTF::setVersion to work
871 const auto oldPlatform = this->platform;
872 this->platform = PLATFORM_PC;
873 switch (newPlatform) {
874 case PLATFORM_UNKNOWN:
875 case PLATFORM_PC:
876 break;
877 case PLATFORM_XBOX:
878 // Have to do it this roundabout way to fix cubemaps, v7.5 has 6 faces
879 this->setVersion(5);
880 this->platform = newPlatform;
881 // It's safe to do this, and we can't use VTF::setVersion now that platform is set
882 this->version = 2;
883 break;
884 case PLATFORM_X360:
885 case PLATFORM_PS3_ORANGEBOX:
886 this->setVersion(4);
887 break;
888 case PLATFORM_PS3_PORTAL2:
889 this->setVersion(5);
890 break;

896 this->removeFlags(FLAG_MASK_XBOX);
897 }
898
899 // XBOX stores thumbnail as single RGB888 pixel, but we assume thumbnail is DXT1 on other platforms
900 if (this->hasThumbnailData()) {

905 this->thumbnailWidth = 0;

907 }
908 }
909
910 // Add/remove fallback data for XBOX
911 if (this->platform != PLATFORM_XBOX && this->hasFallbackData()) {

913 } else if (this->platform == PLATFORM_XBOX) {
914 this->computeFallback();
915 }
916

918
919 if (this->platform != PLATFORM_PC) {

1142void VTF::computeMips(ImageConversion::ResizeFilter filter) {
1143 auto* imageResource = this->getResourceInternal(Resource::TYPE_IMAGE_DATA);
1144 if (!imageResource || !this->hasImageData()) {
1145 return;
1146 }
1147
1148 if (this->mipCount <= 1) {
1149 if (!this->setRecommendedMipCount() || this->mipCount <= 1) {
1150 return;
1151 }
1152 }
1153
1154 auto* outputDataPtr = imageResource->data.data();
1155 const auto faceCount = this->getFaceCount();
1156 const bool compressed = ImageFormatDetails::compressed(this->format);
1157
1158#ifdef SOURCEPP_BUILD_WITH_THREADS
1159 std::vector<std::future<void>> futures;
1160 futures.reserve(this->frameCount * faceCount * this->depth);
1161#endif
1162 for (int j = 0; j < this->frameCount; j++) {
1163 for (int k = 0; k < faceCount; k++) {
1164#ifdef SOURCEPP_BUILD_WITH_THREADS
1165 futures.push_back(std::async(std::launch::async, [this, filter, outputDataPtr, faceCount, compressed, j, k] {
1166#endif
1167 for (int i = 1; i < this->mipCount; i++) {
1168 const auto [mipWidth, mipHeight, mipDepth] = ImageDimensions::getMipDims(i, compressed, this->width, this->height, this->depth);
1169 const auto [mipWidthM1, mipHeightM1] = ImageDimensions::getMipDims(i - 1, compressed, this->width, this->height);
1170 for (int l = 0; l < mipDepth; l++) {

1277void VTF::computeReflectivity() {
1278 static constexpr auto getReflectivityForImage = [](const VTF& vtf, uint16_t frame, uint8_t face, uint16_t slice) {
1279 static constexpr auto getReflectivityForPixel = [](const ImagePixel::RGBA8888* pixel) -> math::Vec3f {
1280 // http://markjstock.org/doc/gsd_talk_11_notes.pdf page 11
1281 math::Vec3f ref{static_cast<float>(pixel->r), static_cast<float>(pixel->g), static_cast<float>(pixel->b)};
1282 ref /= 255.f * 0.9f;
1283 ref[0] *= ref[0];
1284 ref[1] *= ref[1];
1285 ref[2] *= ref[2];
1286 return ref;
1287 };
1288
1289 auto rgba8888Data = vtf.getImageDataAsRGBA8888(0, frame, face, slice);
1290 math::Vec3f out{};
1291 for (uint64_t i = 0; i < rgba8888Data.size(); i += 4) {
1292 out += getReflectivityForPixel(reinterpret_cast<ImagePixel::RGBA8888*>(rgba8888Data.data() + i));
1293 }
1294 return out / (rgba8888Data.size() / sizeof(ImagePixel::RGBA8888));
1295 };
1296
1297 const auto faceCount = this->getFaceCount();
1298
1299#ifdef SOURCEPP_BUILD_WITH_THREADS
1300 if (this->frameCount > 1 || faceCount > 1 || this->depth > 1) {
1301 std::vector<std::future<math::Vec3f>> futures;
1302 futures.reserve(this->frameCount * faceCount * this->depth);
1303
1304 this->reflectivity = {};
1305 for (int j = 0; j < this->frameCount; j++) {
1306 for (int k = 0; k < faceCount; k++) {
1307 for (int l = 0; l < this->depth; l++) {
1308 futures.push_back(std::async(std::launch::async, [this, j, k, l] {
1309 return getReflectivityForImage(*this, j, k, l);
1310 }));
1311 if (futures.size() >= std::thread::hardware_concurrency()) {
1312 for (auto& future : futures) {
1313 this->reflectivity += future.get();
1314 }
1315 futures.clear();
1316 }
1317 }
1318 }
1319 }
1320
1321 for (auto& future : futures) {
1322 this->reflectivity += future.get();
1323 }
1324 this->reflectivity /= this->frameCount * faceCount * this->depth;
1325 } else {
1326 this->reflectivity = getReflectivityForImage(*this, 0, 0, 0);
1327 }
1328#else
1329 this->reflectivity = {};
1330 for (int j = 0; j < this->frameCount; j++) {
1331 for (int k = 0; k < faceCount; k++) {
1332 for (int l = 0; l < this->depth; l++) {
1333 this->reflectivity += getReflectivityForImage(*this, j, k, l);
1334 }
1335 }
1336 }
1337 this->reflectivity /= this->frameCount * faceCount * this->depth;
1338#endif
1339}

1397 std::memcpy(resource->data.data(), data_.data(), data_.size());
1398 return;
1399 }
1400
1401 // Store resource data
1402 std::unordered_map<Resource::Type, std::pair<std::vector<std::byte>, uint64_t>> resourceData;
1403 for (const auto& [type_, flags_, dataSpan] : this->resources) {
1404 resourceData[type_] = {std::vector<std::byte>{dataSpan.begin(), dataSpan.end()}, 0};
1405 }
1406
1407 // Set new resource
1408 if (data_.empty()) {
1409 resourceData.erase(type);
1410 } else {
1411 resourceData[type] = {{data_.begin(), data_.end()}, 0};
1412 }
1413
1414 // Save the data
1415 this->data.clear();
1416 BufferStream writer{this->data};
1417
1418 for (auto resourceType : resourceData | std::views::keys) {
1419 if (!resourceData.contains(resourceType)) {
1420 continue;
1421 }

1424 Resource newResource{
1425 type,
1426 specificResourceData.size() <= sizeof(uint32_t) ? Resource::FLAG_LOCAL_DATA : Resource::FLAG_NONE,
1427 {this->data.data() + offset, specificResourceData.size()},
1428 };
1429 if (auto* resourcePtr = this->getResourceInternal(type)) {
1430 *resourcePtr = newResource;
1431 } else {
1432 this->resources.push_back(newResource);
1433 }
1434 } else if (!resourceData.contains(resourceType)) {
1435 continue;
1436 }
1437 offset = writer.tell();
1438 writer.write(specificResourceData);
1439 }
1440 this->data.resize(writer.size());
1441
1442 for (auto& [type_, flags_, dataSpan] : this->resources) {
1443 if (resourceData.contains(type_)) {
1444 const auto& [specificResourceData, offset] = resourceData[type_];

1446 }

1485 for (int k = 0; k < newFaceCount; k++) {

1488 auto imageSpan = this->getImageDataRaw(i, j, k, l);
1489 std::vector<std::byte> image{imageSpan.begin(), imageSpan.end()};
1490 if (this->width != newWidth || this->height != newHeight) {

1492 }

1494 std::memcpy(newImageData.data() + offset, image.data(), length);
1495 }
1496 }
1497 }
1498 }
1499 }
1500 }
1501 if (this->format != newFormat) {
1502 newImageData = ImageConversion::convertSeveralImageDataToFormat(newImageData, this->format, newFormat, newMipCount, newFrameCount, newFaceCount, newWidth, newHeight, newDepth, quality);
1503 }
1504 }
1505 } else {
1506 newImageData.resize(ImageFormatDetails::getDataLength(newFormat, newMipCount, newFrameCount, newFaceCount, newWidth, newHeight, newDepth));
1507 }
1508
1509 this->format = newFormat;
1510 this->width = newWidth;
1511 this->height = newHeight;
1512 this->mipCount = newMipCount;
1513 this->frameCount = newFrameCount;
1514 if (newFaceCount > 1) {
1515 this->flags |= FLAG_V0_ENVMAP;
1516 } else {
1517 this->flags &= ~FLAG_V0_ENVMAP;
1518 }
1519 this->depth = newDepth;
1520
1521 this->setResourceInternal(Resource::TYPE_IMAGE_DATA, newImageData);
1522
1523 // Fix up XBOX resources that depend on image resource
1524 if (const auto* palette = this->getResource(Resource::TYPE_PALETTE_DATA)) {
1525 const auto targetSize = 256 * sizeof(ImagePixel::BGRA8888) * this->frameCount;
1526 if (palette->data.size() != targetSize) {
1527 std::vector<std::byte> paletteData{palette->data.begin(), palette->data.end()};
1528 paletteData.resize(targetSize);
1529 this->setResourceInternal(Resource::TYPE_PALETTE_DATA, paletteData);
1530 }
1531 }
1532 if (this->hasFallbackData()) {
1533 this->removeFallback();

1535 }
1536}

1547 spriteHeight = 0;
1548
1549 const auto shtResource = this->getResource(Resource::TYPE_PARTICLE_SHEET_DATA);
1550 if (!shtResource) {
1551 return {};
1552 }
1553
1554 auto sht = shtResource->getDataAsParticleSheet();

1569
1570 if (x1 > x2) [[unlikely]] {
1571 std::swap(x1, x2);
1572 }
1573 if (y1 > y2) [[unlikely]] {
1574 std::swap(y1, y2);
1575 }

1994uint64_t VTF::estimateBakeSize(bool& isExact) const {
1995 isExact = true;
1996 uint64_t vtfSize = 0;
1997
1998 switch (this->platform) {
1999 case PLATFORM_UNKNOWN:
2000 break;
2001 case PLATFORM_PC:
2002 switch (this->version) {
2003 case 6:
2004 if (this->compressionLevel > 0) {
2005 vtfSize += sizeof(uint64_t); // AXC header size
2006 vtfSize += sizeof(uint32_t) * (2 + this->mipCount * this->frameCount * this->getFaceCount()); // AXC resource size
2007 }
2008 case 5:
2009 case 4:
2010 case 3:
2011 vtfSize += 11 + sizeof(uint32_t) + sizeof(uint64_t) * this->getResources().size(); // resources header size
2012 for (const auto& [resourceType, resourceFlags, resourceData] : this->getResources()) {
2013 if (!(resourceFlags & Resource::FLAG_LOCAL_DATA) && resourceType != Resource::TYPE_THUMBNAIL_DATA && resourceType != Resource::TYPE_IMAGE_DATA) {
2014 vtfSize += resourceData.size(); // resource size
2015 }
2016 }
2017 case 2:
2018 vtfSize += sizeof(uint16_t); // depth header field
2019 case 1:
2020 case 0:
2021 vtfSize += sizeof(uint32_t) * 9 + sizeof(float) * 4 + sizeof(uint16_t) * 4 + sizeof(uint8_t) * 3; // header fields
2022 if (this->version < 3) {
2023 vtfSize += math::paddingForAlignment(16, vtfSize); // align to 16 bytes
2024 }
2025
2026 if (const auto* thumbnailResource = this->getResource(Resource::TYPE_THUMBNAIL_DATA); thumbnailResource && this->hasThumbnailData()) {
2027 vtfSize += thumbnailResource->data.size(); // thumbnail size
2028 }
2029 if (const auto* imageResource = this->getResource(Resource::TYPE_IMAGE_DATA); imageResource && this->hasImageData()) {
2030 if (this->version < 6 || this->compressionLevel == 0) {
2031 vtfSize += imageResource->data.size(); // uncompressed image size
2032 } else {
2033 isExact = false;
2034 vtfSize += static_cast<uint64_t>(static_cast<double>(imageResource->data.size()) / 1.75); // compressed image data
2035 }
2036 }
2037 break;
2038 default:
2039 break;
2040 }
2041 break;
2042 case PLATFORM_XBOX:
2043 vtfSize += sizeof(uint32_t) * 6 + sizeof(float) * 4 + sizeof(uint16_t) * 6 + sizeof(uint8_t) * 6; // header fields
2044
2045 if (const auto* thumbnailResource = this->getResource(Resource::TYPE_THUMBNAIL_DATA); thumbnailResource && this->hasThumbnailData()) {
2046 vtfSize += thumbnailResource->data.size(); // thumbnail size
2047 }
2048 if (this->format == ImageFormat::P8) {
2049 if (const auto* paletteResource = this->getResource(Resource::TYPE_PALETTE_DATA)) {
2050 vtfSize += paletteResource->data.size(); // palette size
2051 }
2052 }
2053 if (const auto* fallbackResource = this->getResource(Resource::TYPE_FALLBACK_DATA); fallbackResource && this->hasFallbackData()) {
2054 vtfSize += ImageFormatDetails::getDataLengthXBOX(false, this->format, this->fallbackMipCount, this->frameCount, this->getFaceCount(), this->fallbackWidth, this->fallbackHeight); // fallback size
2055 vtfSize += math::paddingForAlignment(512, vtfSize); // align to 512 bytes if fallback present
2056 }
2057 if (const auto* imageResource = this->getResource(Resource::TYPE_IMAGE_DATA); imageResource && this->hasImageData()) {
2058 vtfSize += ImageFormatDetails::getDataLengthXBOX(true, this->format, this->mipCount, this->frameCount, this->getFaceCount(), this->width, this->height, this->depth); // image size
2059 }
2060 if (vtfSize > 512) {
2061 vtfSize += math::paddingForAlignment(512, vtfSize); // align to 512 bytes if longer than 512 bytes
2062 }
2063 break;
2064 case PLATFORM_PS3_PORTAL2:
2065 vtfSize += sizeof(uint32_t); // align to 16 bytes
2066 case PLATFORM_X360:
2067 case PLATFORM_PS3_ORANGEBOX:
2068 vtfSize += sizeof(uint32_t) * 7 + sizeof(float) * 4 + sizeof(uint16_t) * 5 + sizeof(uint8_t) * 6; // header fields
2069 vtfSize += sizeof(uint64_t) * this->getResources().size(); // resources header size
2070
2071 for (const auto& [resourceType, resourceFlags, resourceData] : this->getResources()) {
2072 if (!(resourceFlags & Resource::FLAG_LOCAL_DATA) && resourceType != Resource::TYPE_IMAGE_DATA) {
2073 vtfSize += resourceData.size(); // resource size
2074 }
2075 }
2076 if (const auto* imageResource = this->getResource(Resource::TYPE_IMAGE_DATA); imageResource && this->hasImageData()) {
2077 if (this->compressionMethod != CompressionMethod::CONSOLE_LZMA) {
2078 vtfSize += imageResource->data.size(); // uncompressed image data
2079 } else {
2080 isExact = false;
2081 vtfSize += static_cast<uint64_t>(static_cast<double>(imageResource->data.size()) / 1.75); // compressed image data
2082 }
2083 }
2084 break;
2085 }
2086 return vtfSize;
2087}

2089std::vector<std::byte> VTF::bake() const {
2090 std::vector<std::byte> out;
2091 BufferStream writer{out};
2092
2093 const auto writeResources = [&writer](uint64_t headerLengthPos, const std::vector<Resource>& sortedResources) -> uint64_t {
2094 auto resourceHeaderCurPos = writer.tell();
2095 writer.pad<uint64_t>(sortedResources.size());
2096 auto resourceDataCurPos = writer.tell();
2097 writer.seek_u(headerLengthPos).write<uint32_t>(resourceDataCurPos);
2098
2099 uint64_t resourceHeaderImagePos = 0;
2100 for (auto& resource : sortedResources) {
2101 writer.seek_u(resourceHeaderCurPos);
2102
2103 uint32_t resourceType = resource.type;
2104 if (resource.flags & Resource::FLAG_LOCAL_DATA) {
2105 resourceType |= Resource::FLAG_LOCAL_DATA << 24;
2106 }
2107 if (writer.is_big_endian()) {
2108 // type threeCC is little-endian
2109 BufferStream::swap_endian(&resourceType);
2110 }
2111 writer.write<uint32_t>(resource.type);
2112
2113 if (resource.type == Resource::TYPE_IMAGE_DATA) {
2114 resourceHeaderImagePos = writer.tell();
2115 writer.write<uint32_t>(0);
2116 continue;
2117 }
2118
2119 if (resource.flags & Resource::FLAG_LOCAL_DATA) {
2120 writer.write(resource.data);
2121 resourceHeaderCurPos = writer.tell();
2122 } else {
2123 writer.write<uint32_t>(resourceDataCurPos);
2124 resourceHeaderCurPos = writer.tell();
2125 writer.seek_u(resourceDataCurPos).write(resource.data);
2126 resourceDataCurPos = writer.tell();
2127 }
2128 }
2129 if (resourceHeaderImagePos) {
2130 writer.seek_u(resourceHeaderImagePos).write(resourceDataCurPos);
2131 for (auto& resource : sortedResources) {
2132 if (resource.type == Resource::TYPE_IMAGE_DATA) {
2133 writer.seek_u(resourceDataCurPos).write(resource.data);
2134 break;
2135 }
2136 }
2137 return resourceDataCurPos;
2138 }
2139 return 0;
2140 };
2141
2142 // HACK: no source game supports this format, but they do support the flag with reg. DXT1
2143 auto bakeFormat = this->format;
2144 auto bakeFlags = this->flags;
2145 if (bakeFormat == ImageFormat::DXT1_ONE_BIT_ALPHA) {
2146 bakeFormat = ImageFormat::DXT1;
2147 bakeFlags |= FLAG_V0_ONE_BIT_ALPHA;
2148 }
2149 // HACK: NV_NULL / ATI2N / ATI1N are at a different position based on engine branch
2150 if (this->version < 5 && (this->format == ImageFormat::EMPTY || this->format == ImageFormat::ATI2N || this->format == ImageFormat::ATI1N)) {
2151 bakeFormat = static_cast<ImageFormat>(static_cast<int32_t>(this->format) + 3);
2152 }
2153
2154 switch (this->platform) {
2155 case PLATFORM_UNKNOWN:
2156 return out;
2157 case PLATFORM_PC: {
2158 writer
2159 .write(VTF_SIGNATURE)
2160 .write<int32_t>(7)
2161 .write(this->version);
2162
2163 const auto headerLengthPos = writer.tell();
2164 writer.write<uint32_t>(0);
2165
2166 writer
2167 .write(this->width)
2168 .write(this->height)
2169 .write(bakeFlags)
2170 .write(this->frameCount)
2171 .write(this->startFrame)
2172 .pad<uint32_t>()
2173 .write(this->reflectivity)
2174 .pad<uint32_t>()
2175 .write(this->bumpMapScale)
2176 .write(bakeFormat)
2177 .write(this->mipCount)
2178 .write(this->thumbnailFormat)
2179 .write(this->thumbnailWidth)
2180 .write(this->thumbnailHeight);
2181
2182 if (this->version >= 2) {
2183 writer << this->depth;
2184 }
2185
2186 if (this->version < 3) {
2187 writer.pad(math::paddingForAlignment(16, writer.tell()));
2188 const auto headerSize = writer.tell();
2189 writer.seek_u(headerLengthPos).write<uint32_t>(headerSize).seek_u(headerSize);
2190
2191 if (const auto* thumbnailResource = this->getResource(Resource::TYPE_THUMBNAIL_DATA); thumbnailResource && this->hasThumbnailData()) {
2192 writer.write(thumbnailResource->data);
2193 }
2194 if (const auto* imageResource = this->getResource(Resource::TYPE_IMAGE_DATA); imageResource && this->hasImageData()) {
2195 writer.write(imageResource->data);
2196 }
2197 } else {
2198 std::vector<std::byte> auxCompressionResourceData;
2199 std::vector<std::byte> compressedImageResourceData;
2200 bool hasAuxCompression = false;
2201 if (const auto* imageResource = this->getResource(Resource::TYPE_IMAGE_DATA)) {
2202 hasAuxCompression = this->version >= 6 && this->compressionLevel != 0;
2203 if (hasAuxCompression) {
2204 const auto faceCount = this->getFaceCount();
2205 auxCompressionResourceData.resize((this->mipCount * this->frameCount * faceCount + 2) * sizeof(uint32_t));

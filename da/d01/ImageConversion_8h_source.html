429template<ImagePixel::PixelType P>
430[[nodiscard]] std::vector<std::byte> extractChannelFromImageData(std::span<const std::byte> imageData, auto P::*channel) {
431 using C = sourcepp::member_type_t<decltype(channel)>;
432 if (imageData.empty() || imageData.size() % sizeof(P) != 0) {
433 return {};
434 }
435
436 std::span pixels{reinterpret_cast<const P*>(imageData.data()), imageData.size() / sizeof(P)};
437
438 std::vector<std::byte> out(imageData.size() / sizeof(P) * sizeof(C));
439 BufferStream stream{out, false};
440 for (const auto& pixel : pixels) {
441 stream << pixel.*channel;
442 }
443 return out;
444}

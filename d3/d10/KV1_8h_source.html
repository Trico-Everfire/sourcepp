128 static void read(BufferStreamReadOnly& stream, BufferStream& backing, std::vector<K>& elements, const sourcepp::parser::text::EscapeSequenceMap& escapeSequences) {
129 using namespace sourcepp;
130 while (true) {
131 // Check if the block is over
132 parser::text::eatWhitespaceAndSingleLineComments(stream);
133 if (stream.peek<char>() == '}') {
134 stream.skip();
135 break;
136 }
137 // Read key
138 {
139 auto childKey = parser::text::readStringToBuffer(stream, backing, parser::text::DEFAULT_STRING_START, parser::text::DEFAULT_STRING_END, escapeSequences);
140 elements.push_back(K{});
141 elements.back().key = childKey;
142 parser::text::eatWhitespaceAndSingleLineComments(stream);
143 }
144 // Read value
145 if (stream.peek<char>() != '{') {
146 elements.back().value = parser::text::readStringToBuffer(stream, backing, parser::text::DEFAULT_STRING_START, parser::text::DEFAULT_STRING_END, escapeSequences);
147 parser::text::eatWhitespaceAndSingleLineComments(stream);
148 }
149 // Read conditional
150 if (stream.peek<char>() == '[') {
151 elements.back().conditional = parser::text::readStringToBuffer(stream, backing, "[", "]", escapeSequences);
152 parser::text::eatWhitespaceAndSingleLineComments(stream);
153 }
154 // Read block
155 if (stream.peek<char>() == '{') {
156 stream.skip();
157 parser::text::eatWhitespaceAndSingleLineComments(stream);
158 if (stream.peek<char>() != '}') {
159 KV1ElementBase<S, K>::read(stream, backing, elements.back().children, escapeSequences);
160 } else {
161 stream.skip();
162 }
163 }
164 }
165 }
166
167 S key = ""; // NOLINT(*-redundant-string-init)
168 S value = ""; // NOLINT(*-redundant-string-init)
169 S conditional = ""; // NOLINT(*-redundant-string-init)
170 std::vector<K> children;
171};

274 using namespace sourcepp;
275 constexpr auto writeIndentation = [](BufferStream& stream_, unsigned short indentLevel_) {
276 for (unsigned short i = 0; i < indentLevel_; i++) {
277 stream_.write('\t');
278 }
279 };
280 constexpr auto writeQuotedString = [](BufferStream& stream_, std::string_view str, const parser::text::EscapeSequenceMap& escapeSequences_, char quoteStart = '\"', char quoteEnd = '\"') {
281 stream_.write(quoteStart);
282 if (!str.empty()) {
283 stream_.write(parser::text::convertSpecialCharsToEscapes(str, escapeSequences_), false);
284 }
285 stream_.write(quoteEnd);
286 };
287
288 for (auto& elem : elements) {
289 writeIndentation(stream, indentLevel);
290 writeQuotedString(stream, elem.key, escapeSequences);
291 if (!elem.value.empty() || elem.children.empty()) {
292 stream.write(' ');
293 writeQuotedString(stream, elem.value, escapeSequences);
294 }
295 if (!elem.conditional.empty()) {
296 stream.write(' ');
297 writeQuotedString(stream, elem.conditional, escapeSequences, '[', ']');
298 }
299 stream.write('\n');
300 if (!elem.children.empty()) {
301 writeIndentation(stream, indentLevel);
302 stream << '{' << '\n';
303 write(stream, elem.children, indentLevel + 1, escapeSequences);
304 writeIndentation(stream, indentLevel);
305 stream << '}' << '\n';
306 }
